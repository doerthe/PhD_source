% Set epigraph style
\renewcommand{\epigraph}[2]{%
 \vspace{\baselineskip}%
 % Extend epigraph into the margin
 \begin{adjustwidth}{}{-\marginnotewidth}%
  % Align the epigraph box to the right
  \begin{flushright}%
   % Give the epigraph the width of the preceding chapter title
   \begin{minipage}[t]{\marginnotewidth}%
    \sffamily #1\\[-.9\baselineskip]
    \flushright\itshape\small---~#2%
   \end{minipage}%
  \end{flushright}%
 \end{adjustwidth}%
 \vspace{2\baselineskip}%
}
